# 系统 System

# 获取设备信息 getDeviceInfo

API 介绍

API 原型

DYBridge.dispatch('System','getDeviceInfo',json,function(error,msg,data){})

json参数说明

无

回调参数说明

{
	"uuid":"",//String类型，uuid
    "deviceType":"",//String类型 厂商
    "deviceVersion":"",//String 类型 具体型号
    "systemName":"",//String 类型 系统名称
    "systemVersion":""//String类型 版本号
}

示例

DYBridge.dispatch('System','getDeviceInfo',{},function(error,msg,data){

})

# 设置剪切板 copyText

API 介绍

API 原型

DYBridge.dispatch('System','copyText',json,function(error,msg,data){})

json参数说明

{
    "text":""//String类型 要复制到剪切板的内容
}

回调参数说明

无

示例

DYBridge.dispatch('System','copyText',{
    "text":"复制内容"
},null)

# 判断应用是否安装 isInstall

API 介绍

API 原型

DYBridge.dispatch('System','isInstall',json,function(error,msg,data){})

json参数说明

{
    "package":"", //String类型 Android端为应用包名
    "urlScheme":"" //String类型 iOS 端为应用 urlScheme
}

回调参数说明

{
	"install"：true //bool类型，是否已安装
}

error 为1表示App未安装

示例

DYBridge.dispatch('System','isInstall',{
    "package":"air.tv.douyu.android"
},function(error,msg,data){
	console.log(data.install)
})

# 获取屏幕宽高 getScreenSize

API 介绍

获取屏幕宽高

API 原型

DYBridge.dispatch('System','getScreenSize',arguments,callback)

参数说明

无

回调参数说明

data 为序列化的 json 字符串

{
    "screenWidth": IntValue, // 宽
    "screenHeight": IntValue // 高
}

示例

DYBridge.dispatch('System','getScreenSize',null,funtion(error,msg,data){
    if(error==0){
        console.log(data)
    }
})
